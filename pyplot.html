from collections import Counter
import matplotlib.pyplot as plt

# Provided text
TEXT = ("At three oclock precisely I was at Baker Street, but Holmes had not yet returned. "
        "The landlady informed me that he had left the house shortly after eight oclock in the morning. "
        "I sat down beside the fire, however, with the intention of awaiting him, however long he might be. "
        "I was already deeply interested in his inquiry, for, though it was surrounded by none of the grim and strange features "
        "which were associated with the two crimes which I have already recorded, still, the nature of the case and the exalted "
        "station of his client gave it a character of its own. Indeed, apart from the nature of the investigation which my friend "
        "had on hand, there was something in his masterly grasp of a situation, and his keen, incisive reasoning, which made it a "
        "pleasure to me to study his system of work, and to follow the quick, subtle methods by which he disentangled the most "
        "inextricable mysteries. So accustomed was I to his invariable success that the very possibility of his failing had ceased "
        "to enter into my head.").lower()

# Count character frequencies
letter_freq = dict(Counter(TEXT))

# Sort the dictionary by frequency
sorted_freq = dict(sorted(letter_freq.items(), key=lambda item: item[1], reverse=True))

# Create the horizontal bar plot
plt.figure(figsize=(12, 10))
plt.barh(list(sorted_freq.keys()), list(sorted_freq.values()), color='tan')
plt.xlabel('Frequency')
plt.ylabel('Character')
plt.title('Character Frequency Distribution in Text')
plt.gca().invert_yaxis()  # Highest frequency on top
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import random

# Generate a sequence of at least 5 random integers
random_numbers = [random.randint(1, 100) for _ in range(7)]
indices = list(range(len(random_numbers)))

# Create subplots
fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(8, 6))

# Scatter plot
ax1.scatter(indices, random_numbers, color='blue')
ax1.set_title('Scatter Plot of Random Numbers')
ax1.set_xlabel('Index')
ax1.set_ylabel('Value')

# Line plot
ax2.plot(indices, random_numbers, marker='o', color='green')
ax2.set_title('Line Plot of Random Numbers')
ax2.set_xlabel('Index')
ax2.set_ylabel('Value')

# Adjust layout and display the plots
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import numpy as np

# Generate a sequence of 7 random integers
np.random.seed(0)
data = np.random.randint(1, 100, size=7)
indices = np.arange(len(data))

# Create subplots
fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(8, 6))

# Scatter plot with hexagon markers in dark blue
ax1.scatter(indices, data, marker='h', color='darkblue')
ax1.set_title("Scatter Plot of Random Numbers", fontweight='bold')
ax1.set_xlabel("Index", fontweight='bold')
ax1.set_ylabel("Value", fontweight='bold')

# Line plot with hexagon markers and line in dark red
ax2.plot(indices, data, marker='h', color='darkred', markerfacecolor='darkred')
ax2.set_title("Line Plot of Random Numbers", fontweight='bold')
ax2.set_xlabel("Index", fontweight='bold')
ax2.set_ylabel("Value", fontweight='bold')

plt.tight_layout()
